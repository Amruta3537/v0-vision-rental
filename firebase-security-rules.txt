// BACKEND INTEGRATION POINT 16: Security Rules
// These rules control access to your Firebase database and storage
// You'll need to add these in the Firebase Console

// Firestore Rules
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Allow users to read all rooms
    match /rooms/{roomId} {
      allow read: if true;
      // Only allow write if the user is authenticated and is the owner
      allow create: if request.auth != null;
      allow update, delete: if request.auth != null && request.auth.uid == resource.data.owner;
    }
    
    // Allow users to manage their own favorites
    match /favorites/{favoriteId} {
      allow read, write: if request.auth != null && 
                          (request.auth.uid == resource.data.userId || 
                           request.auth.uid == request.resource.data.userId);
    }
    
    // Allow users to manage their own reviews
    match /reviews/{reviewId} {
      allow read: if true;
      allow write: if request.auth != null && 
                    (request.auth.uid == resource.data.userId || 
                     request.auth.uid == request.resource.data.userId);
    }
  }
}

// Storage Rules
rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    match /rooms/{userId}/{allPaths=**} {
      // Allow users to read all room images
      allow read: if true;
      // Only allow write if the user is authenticated and is the owner
      allow write: if request.auth != null && request.auth.uid == userId;
    }
  }
}
